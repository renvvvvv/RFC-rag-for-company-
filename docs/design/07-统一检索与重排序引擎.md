### 1.2 核心设计原则

| 原则 | 说明 |
|------|------|
| **权限前置优先** | 预过滤阶段将权限条件推送到 Milvus expr 和 Meilisearch filter，避免无权限数据进入后续计算 |
| **多路互补召回** | Dense(语义) + Sparse(关键词) + Structured(元数据) 三路并行，覆盖同义改写、专有名词、结构化查询全场景 |
| **两阶段重排序** | Stage1 轻量模型快速粗筛(低成本)，Stage2 用户精排模型保障质量(高精度)，平衡延迟与效果 |
| **模态统一处理** | 图片/视频/文档/Excel 的 Chunk 统一参与融合和重排序，返回结果按模态分类组装 |
| **零信任权限校验** | 预过滤(粗粒度) + 后过滤(细粒度) 双阶段校验，确保字段级权限零泄露 |

### 2.2 意图识别

```python
from enum import Enum
from typing import List, Optional, Dict, Any
from pydantic import BaseModel

class QueryIntent(Enum):
    FACTUAL = "factual"           # 事实型: "Q3销售额是多少？"
    SUMMARY = "summary"           # 摘要型: "总结一下这份合同的主要内容"
    COMPARISON = "comparison"     # 比较型: "A产品和B产品的区别是什么？"
    LIST = "list"                 # 列表型: "列出所有涉及安全规范的文件"
    NAVIGATION = "navigation"     # 导航型: "打开第三季度的财报"
    MULTIMODAL = "multimodal"     # 多模态: "找一下产品发布会的现场照片"
    STRUCTURED = "structured"     # 结构化: "销售部工资最高的员工是谁？"

class IntentResult(BaseModel):
    intent: QueryIntent
    confidence: float             # 置信度 0~1
    sub_intents: List[QueryIntent] = []  # 次要意图
    entities: Dict[str, Any]      # 提取的实体
    time_constraints: Optional[Dict[str, str]] = None  # 时间约束

class IntentClassifier:
    """查询意图分类器
    
    实现方式: 规则 + 轻量模型双路
    - 规则引擎: 基于关键词模板快速匹配常见意图
    - 模型分类: 用户可接入自己的意图分类模型(可选)
    """
    
    # 规则模板示例
    RULE_PATTERNS = {
        QueryIntent.FACTUAL: [
            r"^(什么|多少|几|谁|哪里|何时|为什么|怎么)",
            r"(是|为)\s*多少",
            r"(目标|金额|数量|比例|增长率)",
        ],
        QueryIntent.SUMMARY: [
            r"(总结|概括|摘要|概述|简述|要点)",
            r"(主要内容|核心观点|关键信息)",
        ],
        QueryIntent.COMPARISON: [
            r"(区别|差异|对比|比较|vs|versus|和.*相比)",
            r"(A|B|方案[一二]).*(还是|和|与)",
        ],
        QueryIntent.LIST: [
            r"^(列出|列举|有哪些|有什么|所有).*(文件|文档|条款|规定)",
            r"(找|查).*涉及.*的",
        ],
        QueryIntent.MULTIMODAL: [
            r"(照片|图片|图像|截图|视频|录像)",
            r"(找|搜).*?(图|视频|照片)",
        ],
        QueryIntent.STRUCTURED: [
            r"(最高|最低|最大|最小|最多|最少|排名|第[一二三四五六七八九十])",
            r"(平均|总和|总计|合计|统计)",
        ],
    }
    
    def classify(self, query: str, user_context: Dict) -> IntentResult:
        """对查询进行意图分类"""
        query = query.strip()
        
        # 1. 规则匹配
        rule_scores = {}
        for intent, patterns in self.RULE_PATTERNS.items():
            score = 0
            for pattern in patterns:
                if re.search(pattern, query, re.IGNORECASE):
                    score += 1
            rule_scores[intent] = score
        
        # 2. 选取最高分的意图
        best_intent = max(rule_scores, key=rule_scores.get)
        best_score = rule_scores[best_intent]
        
        # 3. 提取实体
        entities = self._extract_entities(query)
        
        # 4. 时间约束解析
        time_constraints = self._extract_time(query)
        
        confidence = min(best_score / 2.0, 1.0)  # 简单归一化
        
        return IntentResult(
            intent=best_intent,
            confidence=confidence,
            sub_intents=[k for k, v in rule_scores.items() 
                        if v > 0 and k != best_intent],
            entities=entities,
            time_constraints=time_constraints
        )
    
    def _extract_entities(self, query: str) -> Dict[str, Any]:
        """简单实体提取（生产环境可用NER模型替代）"""
        entities = {}
        
        # 数字提取
        numbers = re.findall(r'\d+(?:\.\d+)?', query)
        if numbers:
            entities["numbers"] = numbers
        
        # 引号内容提取（文件名/专有名词）
        quotes = re.findall(r'["""]([^"""]+)["""]', query)
        if quotes:
            entities["quoted_names"] = quotes
        
        # 季度/年份
        quarters = re.findall(r'(Q[1-4]|第[一二三四]季度)', query, re.IGNORECASE)
        if quarters:
            entities["quarters"] = quarters
        
        years = re.findall(r'(20\d{2})年?', query)
        if years:
            entities["years"] = years
            
        return entities
    
    def _extract_time(self, query: str) -> Optional[Dict[str, str]]:
        """提取时间约束"""
        time_map = {}
        
        # 相对时间
        if re.search(r'最近|最新|近(一|几)(个|周|月|年)', query):
            time_map["type"] = "relative"
            time_map["range"] = "recent"
        elif re.search(r'(去年|上一年|去年)', query):
            time_map["type"] = "relative"
            time_map["range"] = "last_year"
        elif re.search(r'(今年|本年)', query):
            time_map["type"] = "relative"
            time_map["range"] = "this_year"
        
        # 绝对时间范围
        date_range = re.search(r'(20\d{2}).*?到.*?(20\d{2})', query)
        if date_range:
            time_map["type"] = "absolute"
            time_map["start"] = date_range.group(1)
            time_map["end"] = date_range.group(2)
        
        return time_map if time_map else None
```

### 2.3 查询改写

```python
from typing import List
import re

class QueryRewriter:
    """查询改写器
    
    功能:
    1. 同义扩展: 为关键词添加同义词/近义词
    2. 指代消解: 结合对话历史消解"它/这个/那个"
    3. 查询补全: 为不完整查询补充上下文
    4. 多语言对齐: 中英文混合查询标准化
    5. 时间/数值归一化: "去年"→"2025年", "Q3"→"第三季度"
    """
    
    def __init__(self, synonym_dict: Dict[str, List[str]] = None):
        self.synonym_dict = synonym_dict or {}
        self.dialogue_history: List[Dict] = []
    
    def rewrite(self, query: str, 
                dialogue_history: List[Dict] = None,
                intent: QueryIntent = None) -> "RewriteResult":
        """主改写入口"""
        
        original = query
        rewritten = query
        expansions = []
        
        # 1. 指代消解
        if dialogue_history:
            rewritten = self._resolve_anaphora(rewritten, dialogue_history)
            expansions.append(f"指代消解: {rewritten}")
        
        # 2. 时间归一化
        rewritten = self._normalize_time(rewritten)
        expansions.append(f"时间归一化: {rewritten}")
        
        # 3. 同义扩展（意图感知）
        synonym_queries = self._expand_synonyms(rewritten, intent)
        expansions.append(f"同义扩展: {synonym_queries}")
        
        # 4. 多语言标准化
        rewritten = self._normalize_language(rewritten)
        
        # 5. 结构化查询增强（针对Excel/表格查询）
        if intent == QueryIntent.STRUCTURED:
            structured_hint = self._build_structured_hint(rewritten)
        else:
            structured_hint = None
        
        return RewriteResult(
            original=original,
            rewritten=rewritten,
            synonym_queries=synonym_queries,
            structured_hint=structured_hint,
            expansions=expansions
        )
    
    def _resolve_anaphora(self, query: str, 
                          history: List[Dict]) -> str:
        """指代消解"""
        anaphora_patterns = {
            r"^(它|这个|那个|该).{0,3}": None,
            r"^(这份|这个).{0,3}(文件|文档|合同|报告)": None,
            r"^(上面|前文|刚才).{0,3}(提到|说|指出)": None,
        }
        
        resolved = query
        for pattern in anaphora_patterns:
            if re.search(pattern, query):
                # 从历史中提取最近的主题实体
                if history:
                    last_topic = self._extract_topic(history[-1])
                    if last_topic:
                        resolved = re.sub(pattern, last_topic, resolved, count=1)
                        break
        
        return resolved
    
    def _normalize_time(self, query: str) -> str:
        """时间归一化"""
        from datetime import datetime
        current_year = datetime.now().year
        
        replacements = {
            r'今年': f'{current_year}年',
            r'去年': f'{current_year - 1}年',
            r'前年': f'{current_year - 2}年',
            r'Q1': '第一季度',
            r'Q2': '第二季度',
            r'Q3': '第三季度',
            r'Q4': '第四季度',
        }
        
        result = query
        for pattern, replacement in replacements.items():
            result = re.sub(pattern, replacement, result, flags=re.IGNORECASE)
        
        return result
    
    def _expand_synonyms(self, query: str, 
                         intent: QueryIntent) -> List[str]:
        """同义扩展，生成多个查询变体用于多路召回"""
        variants = [query]
        
        # 简单同义词替换
        for word, synonyms in self.synonym_dict.items():
            if word in query:
                for syn in synonyms[:2]:  # 最多2个同义词
                    variant = query.replace(word, syn)
                    variants.append(variant)
        
        return list(set(variants))
    
    def _normalize_language(self, query: str) -> str:
        """多语言标准化（简单处理，生产环境可用transliteration库）"""
        # 全角转半角
        result = query
        # 中英文标点统一
        result = result.replace('，', ',').replace('。', '.').replace('？', '?')
        return result
    
    def _build_structured_hint(self, query: str) -> Dict[str, Any]:
        """为结构化查询构建提示"""
        hint = {}
        
        # 识别聚合操作
        if re.search(r'(最高|最大|最多)', query):
            hint["aggregate"] = "MAX"
        elif re.search(r'(最低|最小|最少)', query):
            hint["aggregate"] = "MIN"
        elif re.search(r'(平均|均值)', query):
            hint["aggregate"] = "AVG"
        elif re.search(r'(总和|总计|合计)', query):
            hint["aggregate"] = "SUM"
        
        # 识别排序
        if re.search(r'(排名|第[一二三四五六七八九十]|前\d+)', query):
            hint["order_by"] = "DESC"
        
        # 识别目标列
        col_match = re.search(r'(工资|薪资|销售额|销量|人数|数量|金额)', query)
        if col_match:
            hint["target_column_hint"] = col_match.group(1)
        
        return hint


class RewriteResult(BaseModel):
    """查询改写结果"""
    original: str                    # 原始查询
    rewritten: str                   # 改写后主查询
    synonym_queries: List[str]       # 同义扩展查询列表
    structured_hint: Optional[Dict[str, Any]] = None  # 结构化查询提示
    expansions: List[str]            # 改写步骤记录
```

### 2.4 模态路由

```python
from enum import Enum
from typing import Optional

class ModalityRoute(Enum):
    TEXT_ONLY = "text_only"           # 纯文本检索
    IMAGE_SEARCH = "image_search"     # 以图搜图
    MULTIMODAL = "multimodal"         # 文本+图片混合检索
    VIDEO_SEARCH = "video_search"     # 视频检索（关键帧）
    STRUCTURED = "structured"         # 结构化数据检索（Excel）

class ModalityRouter:
    """模态路由决策器
    
    根据查询内容和用户输入决定检索哪些模态的索引集合。
    """
    
    def route(self, query: str, 
              attached_image: Optional[str] = None,
              intent: Optional[QueryIntent] = None) -> ModalityRoute:
        """决定检索路由"""
        
        # 1. 有附件图片 -> 以图搜图 或 多模态
        if attached_image:
            if intent == QueryIntent.MULTIMODAL:
                return ModalityRoute.MULTIMODAL
            return ModalityRoute.IMAGE_SEARCH
        
        # 2. 意图驱动的路由
        if intent == QueryIntent.MULTIMODAL:
            return ModalityRoute.MULTIMODAL
        
        if intent == QueryIntent.STRUCTURED:
            return ModalityRoute.STRUCTURED
        
        # 3. 关键词驱动的路由
        if self._is_image_query(query):
            return ModalityRoute.MULTIMODAL
        
        if self._is_video_query(query):
            return ModalityRoute.VIDEO_SEARCH
        
        # 4. 默认: 纯文本（覆盖文档/Excel/链接等文本类内容）
        return ModalityRoute.TEXT_ONLY
    
    def _is_image_query(self, query: str) -> bool:
        """判断是否为图片相关查询"""
```

## 3. 权限预过滤

### 3.1 预过滤策略

权限预过滤的目标是在**检索执行前**将粗粒度权限条件推送到向量数据库和全文检索引擎，避免无权限数据参与后续计算。

```
权限预过滤层级:

Level 1 (向量库层):  Milvus expr 表达式
    - 文件类型 IN allowed_types
    - doc_id NOT IN denied_docs
    - kb_id IN kb_scope
    - created_at BETWEEN time_range
    
Level 2 (全文检索层): Meilisearch filter
    - tags NOT IN denied_tags
    - tags IN allowed_tags (至少一个)
    - status = "active"
    
Level 3 (组合层): 预过滤后候选集
    - 粗粒度过滤后的候选进入多路召回
```

### 3.2 `build_permission_filter` 实现

```python
from typing import List, Optional, Tuple
from datetime import datetime

class PermissionContext(BaseModel):
    """权限上下文"""
    user_id: str
    allowed_file_types: List[str]
    allowed_doc_ids: Optional[List[str]] = None   # 白名单模式
    denied_doc_ids: List[str] = []                # 黑名单模式
    allowed_tags: List[str] = []
    denied_tags: List[str] = []
    field_permissions: Dict[str, Any] = {}
    time_range: Optional[Tuple[datetime, datetime]] = None
    kb_scope: List[str] = []

class PermissionFilterBuilder:
    """权限过滤器构建器
    
    将 PermissionContext 转换为各检索引擎的过滤表达式。
    """
    
    def build_permission_filter(self, 
                                ctx: PermissionContext) -> "PermissionFilterExpr":
        """构建权限预过滤表达式"""
        
        # 1. 构建 Milvus 表达式
        milvus_expr = self._build_milvus_expr(ctx)
        
        # 2. 构建 Meilisearch 过滤器
        meili_filter = self._build_meilisearch_filter(ctx)
        
        # 3. 构建结构化元数据过滤器
        structured_filter = self._build_structured_filter(ctx)
        
        return PermissionFilterExpr(
            milvus_expr=milvus_expr,
            meilisearch_filter=meili_filter,
            structured_filter=structured_filter,
        )
    
    def _build_milvus_expr(self, ctx: PermissionContext) -> str:
        """构建 Milvus 过滤表达式
        
        Milvus 支持布尔表达式，语法类似 SQL WHERE 子句。
        参考: https://milvus.io/docs/boolean.md
        """
        conditions = []
        
        # 条件1: 知识库范围
        if ctx.kb_scope:
            kb_list = ", ".join([f"'{k}'" for k in ctx.kb_scope])
            conditions.append(f"kb_id in [{kb_list}]")
        
        # 条件2: 文件类型
        if ctx.allowed_file_types:
            type_list = ", ".join([f"'{t}'" for t in ctx.allowed_file_types])
            conditions.append(f"modality in [{type_list}]")
        
        # 条件3: 文档黑名单
        if ctx.denied_doc_ids:
            denied_list = ", ".join([f"'{d}'" for d in ctx.denied_doc_ids])
            conditions.append(f"doc_id not in [{denied_list}]")
        
        # 条件4: 文档白名单（严格模式）
        if ctx.allowed_doc_ids:
            allowed_list = ", ".join([f"'{d}'" for d in ctx.allowed_doc_ids])
            conditions.append(f"doc_id in [{allowed_list}]")
        
        # 条件5: 时间范围
        if ctx.time_range:
            start_ts = int(ctx.time_range[0].timestamp())
            end_ts = int(ctx.time_range[1].timestamp())
            conditions.append(
                f"created_at >= {start_ts} and created_at <= {end_ts}"
            )
        
        # 条件6: 状态
        conditions.append("status == 'active'")
        
        # 组合条件
        if not conditions:
            return ""
        
        return " and ".join(f"({c})" for c in conditions)
    
    def _build_meilisearch_filter(self, ctx: PermissionContext) -> str:
        """构建 Meilisearch 过滤表达式
        
        Meilisearch filter 语法: https://www.meilisearch.com/docs/learn/filtering/filtering
        """
        filters = []
        
        # 知识库范围
        if ctx.kb_scope:
            kb_filters = [f"kb_id = '{k}'" for k in ctx.kb_scope]
            filters.append(f"({' OR '.join(kb_filters)})")
        
        # 文件类型
        if ctx.allowed_file_types:
            type_filters = [f"modality = '{t}'" for t in ctx.allowed_file_types]
            filters.append(f"({' OR '.join(type_filters)})")
        
        # 文档黑名单
        if ctx.denied_doc_ids:
            denied_filters = [f"doc_id != '{d}'" for d in ctx.denied_doc_ids]
            filters.append(f"({' AND '.join(denied_filters)})")
        
        # 标签黑名单
        if ctx.denied_tags:
            denied_tag_filters = [f"tags != '{t}'" for t in ctx.denied_tags]
            filters.append(f"({' AND '.join(denied_tag_filters)})")
        
        # 标签白名单（OR语义: 至少有一个允许标签）
        if ctx.allowed_tags:
            allowed_tag_filters = [f"tags = '{t}'" for t in ctx.allowed_tags]
            filters.append(f"({' OR '.join(allowed_tag_filters)})")
        
        # 状态
        filters.append("status = 'active'")
        
        return " AND ".join(filters)
    
    def _build_structured_filter(self, ctx: PermissionContext) -> Dict[str, Any]:
        """构建结构化元数据过滤器（用于PostgreSQL等关系查询）"""
        filter_dict = {
            "kb_id": {"in": ctx.kb_scope},
            "modality": {"in": ctx.allowed_file_types},
            "status": "active",
        }
        
        if ctx.denied_doc_ids:
            filter_dict["doc_id"] = {"not_in": ctx.denied_doc_ids}
        
        if ctx.allowed_doc_ids:
            filter_dict["doc_id"] = {"in": ctx.allowed_doc_ids}
        
        if ctx.time_range:
            filter_dict["created_at"] = {
                "gte": ctx.time_range[0],
                "lte": ctx.time_range[1],
            }
        
        return filter_dict


class PermissionFilterExpr(BaseModel):
    """权限过滤表达式集合"""
    milvus_expr: str
    meilisearch_filter: str
    structured_filter: Dict[str, Any]
```

### 3.3 预过滤配置参数

| 过滤维度 | 过滤层级 | Milvus 实现 | Meilisearch 实现 | 适用场景 |
|----------|----------|-------------|------------------|----------|
| **文件类型** | Level 1 | `modality in [...]` | `modality = ... OR ...` | 角色-类型权限矩阵 |
| **知识库范围** | Level 1 | `kb_id in [...]` | `kb_id = ... OR ...` | 多租户隔离 |
| **文档黑名单** | Level 1 | `doc_id not in [...]` | `doc_id != ... AND ...` | 显式拒绝的文档 |
| **文档白名单** | Level 1 | `doc_id in [...]` | `doc_id = ... OR ...` | 严格授权模式 |
| **标签黑名单** | Level 2 | - | `tags != ... AND ...` | 敏感标签过滤 |
| **标签白名单** | Level 2 | - | `tags = ... OR ...` | 安全等级标签 |
| **时间范围** | Level 1 | `created_at >= ts` | `created_at >= ts` | 归档文档过滤 |
| **状态过滤** | Level 1 | `status == 'active'` | `status = 'active'` | 排除废弃Chunk |

### 4.3 并行多路召回实现

```python
import asyncio
from typing import List, Dict, Any, Optional
from dataclasses import dataclass

@dataclass
class RetrievalResult:
    """单路召回结果"""
    source: str                    # 来源路径标识
    candidates: List[Candidate]    # 候选列表
    latency_ms: float             # 延迟
    is_fallback: bool = False     # 是否为降级结果

@dataclass
class Candidate:
    """召回候选"""
    chunk_id: str
    doc_id: str
    score: float                  # 该路径的原始分数
    rank: int                     # 在该路径中的排名
    content: str
    metadata: Dict[str, Any]
    source_collection: str        # 来源Collection
    dense_vector: Optional[List[float]] = None
    sparse_vector: Optional[Dict[str, float]] = None

class MultiWayRetriever:
    """多路召回器"""
    
    def __init__(self,
                 milvus_client,
                 meilisearch_client,
                 postgres_client,
                 embedding_service,
                 sparse_encoder):
        self.milvus = milvus_client
        self.meili = meilisearch_client
        self.pg = postgres_client
        self.embed = embedding_service
        self.sparse = sparse_encoder
    
    async def retrieve(self,
                       query: str,
                       query_vector: List[float],
                       permission_filter: PermissionFilterExpr,
                       modality_route: ModalityRoute,
                       config: Optional[Dict] = None) -> List[RetrievalResult]:
        """并行多路召回"""
        
        config = config or {}
        tasks = []
        
        # 1. Dense 检索任务
        if self._should_run_dense(modality_route):
            tasks.append(self._dense_retrieval(
                query_vector=query_vector,
                milvus_expr=permission_filter.milvus_expr,
                collections=self._get_dense_collections(modality_route),
                top_k=config.get("dense_top_k", 200),
            ))
        
        # 2. Sparse 检索任务
        if self._should_run_sparse(modality_route):
            tasks.append(self._sparse_retrieval(
                query_text=query,
                meili_filter=permission_filter.meilisearch_filter,
                top_k=config.get("sparse_top_k", 200),
            ))
        
        # 3. Structured 检索任务
        if self._should_run_structured(modality_route):
            tasks.append(self._structured_retrieval(
                query_text=query,
                structured_filter=permission_filter.structured_filter,
                top_k=config.get("structured_top_k", 100),
            ))
        
        # 4. Visual 检索任务（以图搜图场景）
        if modality_route in (ModalityRoute.IMAGE_SEARCH, ModalityRoute.MULTIMODAL):
            tasks.append(self._visual_retrieval(
                query_vector=query_vector,  # 视觉向量
                milvus_expr=permission_filter.milvus_expr,
                top_k=config.get("visual_top_k", 50),
            ))
        
        # 并行执行所有召回任务
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # 处理结果和异常
        valid_results = []
        for i, result in enumerate(results):
            if isinstance(result, Exception):
                # 记录错误并返回空结果
                logger.error(f"Retrieval path {i} failed: {result}")
                valid_results.append(RetrievalResult(
                    source=f"path_{i}",
                    candidates=[],
                    latency_ms=0,
                    is_fallback=True
                ))
            else:
                valid_results.append(result)
        
        return valid_results
    
    async def _dense_retrieval(self,
                               query_vector: List[float],
                               milvus_expr: str,
                               collections: List[str],
                               top_k: int) -> RetrievalResult:
        """Dense 向量检索"""

        all_candidates = []
        
        for collection in collections:
            try:
                results = self.milvus.search(
                    collection_name=collection,

                )
                
                for hits in results:

                        candidate = Candidate(
                            chunk_id=hit["entity"]["chunk_id"],
                            doc_id=hit["entity"]["doc_id"],
                            score=hit["distance"],
                            rank=idx + 1,
                            content=hit["entity"]["content"],
                            metadata=hit["entity"]["metadata"],
                            source_collection=collection,
                        )
                        all_candidates.append(candidate)
                        
            except Exception as e:
                logger.error(f"Dense search failed for {collection}: {e}")
        
        # 按分数排序并截断
        all_candidates.sort(key=lambda x: x.score, reverse=True)
        all_candidates = all_candidates[:top_k]
        
        # 重新赋rank
        for i, c in enumerate(all_candidates):
            c.rank = i + 1

        return RetrievalResult(

        )
    
    async def _visual_retrieval(self,

                logger.error(f"Visual search failed for {collection}: {e}")
        
        all_candidates.sort(key=lambda x: x.score, reverse=True)
        all_candidates = all_candidates[:top_k]
        
        latency = (time.time() - start) * 1000
        return RetrievalResult(
            source="visual",
            candidates=all_candidates,
            latency_ms=latency
        )
    
    def _should_run_dense(self, route: ModalityRoute) -> bool:
        return route != ModalityRoute.IMAGE_SEARCH
    
    def _should_run_sparse(self, route: ModalityRoute) -> bool:
        return route in (ModalityRoute.TEXT_ONLY, ModalityRoute.MULTIMODAL, 
                        ModalityRoute.STRUCTURED)
    
    def _should_run_structured(self, route: ModalityRoute) -> bool:
        return True  # 结构化检索作为补充始终运行
    
    def _get_dense_collections(self, route: ModalityRoute) -> List[str]:
        """根据路由获取目标 Dense Collection"""
        collection_map = {
            ModalityRoute.TEXT_ONLY: ["document_dense", "excel_dense", "link_dense"],
            ModalityRoute.MULTIMODAL: ["document_dense", "excel_dense", 
                                        "image_text", "video_keyframe", "link_dense"],
            ModalityRoute.STRUCTURED: ["excel_dense", "document_dense"],
            ModalityRoute.VIDEO_SEARCH: ["video_keyframe", "video_description"],
            ModalityRoute.IMAGE_SEARCH: [],  # 以图搜图不走Dense文本
        }
        return collection_map.get(route, ["document_dense"])
```

## 5. 结果融合

### 5.1 RRF 融合算法

Reciprocal Rank Fusion (RRF) 是多路召回结果融合的标准算法。其核心思想是将各路检索的排名转换为分数后加权求和，不依赖原始分数的绝对值（解决不同路径分数不可比的问题）。

```
RRF 公式:

                N
               ───
               ╲     1
  score(d)  =   ╲  ────────
               ╱    k + rᵢ(d)
               ───
               i=1

其中:
  - N: 召回路径数量
  - rᵢ(d): 文档 d 在第 i 条路径中的排名（从1开始）
  - k: 常数，通常取 60

RRF 特点:
  - 排名越靠前（rᵢ越小），贡献越大
  - 未在某路径中出现的文档，该路径贡献为0
  - k=60 是论文推荐的平滑参数
```

### 5.2 RRF 实现代码

```python
from typing import List, Dict, Tuple
from collections import defaultdict
import math

class RRFFusion:
    """Reciprocal Rank Fusion 结果融合器"""
    
    DEFAULT_K = 60  # Cormack et al. 推荐值
    
    def __init__(self, k: int = DEFAULT_K):
        self.k = k
    
    def fuse(self, 
             retrieval_results: List[RetrievalResult],
             weights: Optional[Dict[str, float]] = None) -> List[Candidate]:
        """
        融合多路召回结果
        
        Args:
            retrieval_results: 各路检索结果
            weights: 可选的路径权重 {source_name: weight}
                    默认所有路径权重为 1.0
        
        Returns:
            融合排序后的候选列表
        """
        if not weights:
            weights = {}
        
        # 1. 收集所有候选的 RRF 分数
        rrf_scores: Dict[str, Dict[str, Any]] = defaultdict(
            lambda: {"score": 0.0, "candidate": None, "sources": set()}
        )
        
        for result in retrieval_results:
            if not result.candidates:
                continue
            
            source_weight = weights.get(result.source, 1.0)
            
            for candidate in result.candidates:
                chunk_id = candidate.chunk_id
                
                # RRF 分数计算: weight * (1 / (k + rank))
                rank_contribution = source_weight * (1.0 / (self.k + candidate.rank))
                rrf_scores[chunk_id]["score"] += rank_contribution
                rrf_scores[chunk_id]["sources"].add(result.source)
                
                # 保留最佳候选对象（多路径召回同一chunk时保留分数最高的）
                if (rrf_scores[chunk_id]["candidate"] is None or 
                    candidate.score > rrf_scores[chunk_id]["candidate"].score):
                    rrf_scores[chunk_id]["candidate"] = candidate
        
        # 2. 组装融合结果
        fused_candidates = []
        for chunk_id, data in rrf_scores.items():
            candidate = data["candidate"]
            # 更新融合分数
            candidate.score = data["score"]
            # 记录来源路径（用于调试和可解释性）
            candidate.metadata["fusion_sources"] = list(data["sources"])
            candidate.metadata["fusion_score"] = data["score"]
            fused_candidates.append(candidate)
        
        # 3. 按 RRF 分数降序排列
        fused_candidates.sort(key=lambda x: x.score, reverse=True)
        
        # 4. 重新赋rank
        for i, c in enumerate(fused_candidates):
            c.rank = i + 1
        
        return fused_candidates
    
    def fuse_with_min_sources(self,
                              retrieval_results: List[RetrievalResult],
                              min_sources: int = 2,
                              weights: Optional[Dict[str, float]] = None) -> List[Candidate]:
        """
        增强版 RRF: 只保留被至少 min_sources 条路径召回的候选
        
        这可以提高结果的相关性（被多路同时召回的候选更可靠）。
        """
        all_fused = self.fuse(retrieval_results, weights)
        
        filtered = [
            c for c in all_fused 
            if len(c.metadata.get("fusion_sources", [])) >= min_sources
        ]
        
        # 如果过滤后太少，回退到全部结果
        if len(filtered) < 10:
            return all_fused
        
        return filtered


class WeightedFusion:
    """加权融合（可选方案）
    
    当不同路径的分数经过归一化后可比较时，可使用加权求和。
    适用于需要显式调节某路权重的场景。
    """
    
    def fuse(self,
             retrieval_results: List[RetrievalResult],
             weights: Dict[str, float],
             normalize_method: str = "minmax") -> List[Candidate]:
        """
        加权融合
        
        Args:
            normalize_method: "minmax" | "zscore" | "softmax"
        """
        score_map: Dict[str, Dict[str, Any]] = defaultdict(
            lambda: {"score": 0.0, "candidate": None, "count": 0}
        )
        
        for result in retrieval_results:
            if not result.candidates or result.source not in weights:
                continue
            
            weight = weights[result.source]
            
            # 归一化分数
            scores = [c.score for c in result.candidates]
            normalized = self._normalize(scores, method=normalize_method)
            
            for candidate, norm_score in zip(result.candidates, normalized):
                chunk_id = candidate.chunk_id
                score_map[chunk_id]["score"] += weight * norm_score
                score_map[chunk_id]["count"] += 1
                
                if score_map[chunk_id]["candidate"] is None:
                    score_map[chunk_id]["candidate"] = candidate
        
        # 平均化（被多路召回的候选取平均）
        fused = []
        for chunk_id, data in score_map.items():
            candidate = data["candidate"]
            candidate.score = data["score"] / max(data["count"], 1)
            fused.append(candidate)
        
        fused.sort(key=lambda x: x.score, reverse=True)
        return fused
    
    def _normalize(self, scores: List[float], method: str) -> List[float]:
        """分数归一化"""
        if not scores:
            return []
        
        if method == "minmax":
            min_s, max_s = min(scores), max(scores)
            if max_s == min_s:
                return [1.0] * len(scores)
            return [(s - min_s) / (max_s - min_s) for s in scores]
        
        elif method == "zscore":
            mean = sum(scores) / len(scores)
            std = math.sqrt(sum((s - mean) ** 2 for s in scores) / len(scores))
            if std == 0:
                return [0.0] * len(scores)
            return [(s - mean) / std for s in scores]
        
        elif method == "softmax":
            exp_scores = [math.exp(s) for s in scores]
            sum_exp = sum(exp_scores)
            return [e / sum_exp for e in exp_scores]
        
        return scores
```

### 6.1 重排序架构

```

    │
    ▼

│                                                                             │
```

### 6.2 Stage 1: 轻量预排序 (`bge-reranker-base`)

```python
from typing import List
import torch
from transformers import AutoTokenizer, AutoModelForSequenceClassification

class Stage1Reranker:
    """Stage 1 轻量预排序器
    
    使用 bge-reranker-base 进行快速粗筛。
    特点:
    - 模型轻量 (~100M参数), 推理速度快
    - Cross-Encoder 架构，query-doc 交互充分
    - 支持中英文
    """
    
    def __init__(self, 
                 model_name: str = "BAAI/bge-reranker-base",
                 device: str = "cuda",
                 batch_size: int = 32,
                 max_length: int = 512):
        self.device = device if torch.cuda.is_available() else "cpu"
        self.batch_size = batch_size
        self.max_length = max_length
        
        # 加载模型和tokenizer
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(model_name)
        self.model.to(self.device)
        self.model.eval()
    
    def rerank(self,
               query: str,
               candidates: List[Candidate],
               top_k: int = 50) -> List[Candidate]:
        """
        对候选进行轻量预排序
        
        Args:
            query: 用户查询
            candidates: 融合后的候选列表
            top_k: 保留的Top-K数量
        
        Returns:
            重排序后的候选列表
        """
        if not candidates:
            return []
        
        # 1. 构建输入对 (query, content)
        pairs = [(query, c.content) for c in candidates]
        
        # 2. 分批推理
        all_scores = []
        for i in range(0, len(pairs), self.batch_size):
            batch_pairs = pairs[i:i + self.batch_size]
            scores = self._infer_batch(batch_pairs)
            all_scores.extend(scores)
        
        # 3. 更新候选分数
        for candidate, score in zip(candidates, all_scores):
            candidate.score = float(score)
            candidate.metadata["stage1_score"] = float(score)
        
        # 4. 按分数降序排列
        candidates.sort(key=lambda x: x.score, reverse=True)
        
        # 5. 截断
        result = candidates[:top_k]
        
        # 6. 重新赋rank
        for i, c in enumerate(result):
            c.rank = i + 1
        
        return result
    
    def _infer_batch(self, pairs: List[Tuple[str, str]]) -> List[float]:
        """批量推理"""
        with torch.no_grad():
            inputs = self.tokenizer(
                pairs,
                padding=True,
                truncation=True,
                max_length=self.max_length,
                return_tensors="pt"
            )
            inputs = {k: v.to(self.device) for k, v in inputs.items()}
            
            outputs = self.model(**inputs)
            scores = outputs.logits.squeeze(-1)
            
            # sigmoid 转为概率
            scores = torch.sigmoid(scores)
            
        return scores.cpu().tolist()
```

### 6.3 Stage 2: 精排模型（用户 Re-rank 模型）

```python
class Stage2Reranker:
    """Stage 2 精排模型
    
    使用用户提供的 Re-rank 模型进行高精度排序。
    支持两种模式:
    1. Cross-Encoder 模式: 用户接入自己的 Cross-Encoder 模型
    2. LLM-based 模式: 使用 LLM (如 minimax-m3) 进行列表式排序
    """
    
    def __init__(self,
                 model_type: str = "cross_encoder",  # "cross_encoder" | "llm"
                 model_name: Optional[str] = None,
                 model_endpoint: Optional[str] = None,
                 api_key: Optional[str] = None,
                 batch_size: int = 16,
                 device: str = "cuda"):
        self.model_type = model_type
        self.batch_size = batch_size
        self.device = device
        
        if model_type == "cross_encoder":
            # 用户自有 Cross-Encoder
            self.tokenizer = AutoTokenizer.from_pretrained(model_name)
            self.model = AutoModelForSequenceClassification.from_pretrained(model_name)
            self.model.to(self.device)
            self.model.eval()
        elif model_type == "llm":
            # LLM-based 排序，通过 API 调用
            self.model_endpoint = model_endpoint
            self.api_key = api_key
    
    def rerank(self,
               query: str,
               candidates: List[Candidate],
               top_k: int = 10) -> List[Candidate]:
        """
        Stage 2 精排
        
        Args:
            query: 用户查询
            candidates: Stage 1 输出的候选（约50个）
            top_k: 最终保留的Top-K数量
        """
        if not candidates:
            return []
        
        if self.model_type == "cross_encoder":
            return self._rerank_cross_encoder(query, candidates, top_k)
        else:
            return self._rerank_llm(query, candidates, top_k)
    
    def _rerank_cross_encoder(self,
                              query: str,
                              candidates: List[Candidate],
                              top_k: int) -> List[Candidate]:
        """Cross-Encoder 精排"""
        pairs = [(query, c.content) for c in candidates]
        
        all_scores = []
        for i in range(0, len(pairs), self.batch_size):
            batch_pairs = pairs[i:i + self.batch_size]
            scores = self._infer_cross_encoder(batch_pairs)
            all_scores.extend(scores)
        
        # 更新分数
        for candidate, score in zip(candidates, all_scores):
            candidate.score = float(score)
            candidate.metadata["stage2_score"] = float(score)
        
        # 排序并截断
        candidates.sort(key=lambda x: x.score, reverse=True)
        result = candidates[:top_k]
        
        for i, c in enumerate(result):
            c.rank = i + 1
        
        return result
    
    def _infer_cross_encoder(self, pairs: List[Tuple[str, str]]) -> List[float]:
        """Cross-Encoder 批量推理"""
        with torch.no_grad():
            inputs = self.tokenizer(
                pairs,
                padding=True,
                truncation=True,
                max_length=512,
                return_tensors="pt"
            )
            inputs = {k: v.to(self.device) for k, v in inputs.items()}
            outputs = self.model(**inputs)
            scores = torch.sigmoid(outputs.logits.squeeze(-1))
        return scores.cpu().tolist()
    
    def _rerank_llm(self,
                    query: str,
                    candidates: List[Candidate],
                    top_k: int) -> List[Candidate]:
        """LLM-based 精排（列表式排序 / Pointwise 评分）"""
        
        # 方案A: Pointwise - 逐个打分（更精确，但更慢）
        # 方案B: Listwise - 让LLM直接输出排序（更快，但可能不稳定）
        
        # 这里使用 Pointwise 方案
        all_scores = []
        for candidate in candidates:
            score = self._llm_score(query, candidate)
            all_scores.append(score)
        
        for candidate, score in zip(candidates, all_scores):
            candidate.score = score
            candidate.metadata["stage2_score"] = score
            candidate.metadata["stage2_type"] = "llm"
        
        candidates.sort(key=lambda x: x.score, reverse=True)
        result = candidates[:top_k]
        
        for i, c in enumerate(result):
            c.rank = i + 1
        
        return result
    
    def _llm_score(self, query: str, candidate: Candidate) -> float:
        """使用LLM对单个query-doc对进行相关性评分"""
        prompt = f"""请评估以下查询与文档片段的相关性。

查询: {query}

文档片段: {candidate.content[:400]}

请仅输出一个 0 到 1 之间的数字，表示相关性分数。
0 表示完全不相关，1 表示完全相关。

分数:"""
        
        # 调用 LLM API (minimax-m3 或其他)
        # response = call_llm_api(prompt, self.model_endpoint, self.api_key)
        # score = extract_number(response)
        
        # 示例返回值
        score = 0.85  # 实际应从LLM响应解析
        return score
```

### 6.5 重排序服务封装

```python
class TwoStageReranker:
    """两阶段重排序服务封装"""
    
    def __init__(self,
                 stage1: Stage1Reranker,
                 stage2: Stage2Reranker,
                 enable_stage1: bool = True,
                 enable_stage2: bool = True,
                 fallback_to_stage1: bool = True):
        self.stage1 = stage1
        self.stage2 = stage2
        self.enable_stage1 = enable_stage1
        self.enable_stage2 = enable_stage2
        self.fallback_to_stage1 = fallback_to_stage1
    
    async def rerank(self,
                     query: str,
                     candidates: List[Candidate]) -> List[Candidate]:
        """执行两阶段重排序"""
        
        # Stage 1: 轻量预排序
        if self.enable_stage1 and candidates:
            candidates = self.stage1.rerank(query, candidates, top_k=50)
        
        # Stage 2: 精排
        if self.enable_stage2 and candidates:
            try:
                candidates = self.stage2.rerank(query, candidates, top_k=10)
            except Exception as e:
                logger.error(f"Stage 2 rerank failed: {e}")
                if self.fallback_to_stage1:
                    # 回退: 使用 Stage 1 的结果截断
                    candidates = candidates[:10]
                else:
                    raise
        else:
            # Stage 2 关闭，直接截断 Stage 1 结果
            candidates = candidates[:10] if candidates else []
        
        return candidates
```

### 7.1 后过滤定位

权限预过滤解决了**粗粒度**权限问题（文件类型、文档ID、标签），但无法处理需要精确校验**文档内容位置**的细粒度权限。权限后过滤在重排序后、生成前执行，对进入最终上下文的候选进行**字段级精确校验**。

```
权限过滤双阶段对比:

预过滤 (Pre-filter)              后过滤 (Post-filter)

文档白/黑名单 ✓                  文档级 (已由预过滤覆盖)
标签 ✓                           标签二次确认 ✓
时间范围 ✓                       时间范围 (已由预过滤覆盖)
状态 ✓                           状态 (已由预过滤覆盖)
─────────────────────            ─────────────────────
无法处理:                         核心职责:
  · Excel列级权限                  · Excel列级权限校验
  · Word段落级权限                 · Word段落级权限校验
  · 字段级标签权限                 · 字段级标签权限校验
```

### 7.2 `apply_field_level_filter` 实现

```python
class FieldLevelFilter:
    """字段级权限后过滤器
    
    对 Stage 2 重排序后的候选进行细粒度权限校验。
    逐 Chunk 检查其内容位置是否在用户权限范围内。
    """
    
    def apply_field_level_filter(self,
                                  candidates: List[Candidate],
                                  user: User,
                                  permission_ctx: PermissionContext) -> List[Candidate]:
        """
        应用字段级权限过滤
        
        处理逻辑:
        1. Excel: 检查 involved_columns 是否在用户允许列范围内
        2. Word: 检查段落位置是否在用户允许范围内
        3. 通用: 检查 Chunk 级标签权限
        
        Args:
            candidates: Stage 2 重排序后的候选（Top-10）
            user: 当前用户
            permission_ctx: 权限上下文
        
        Returns:
            过滤后的候选列表
        """
        filtered = []
        
        for candidate in candidates:
            # 获取文档模态
            modality = candidate.metadata.get("modality", "document")
            
            # 根据模态选择过滤策略
            if modality == "excel":
                if self._check_excel_permission(candidate, user, permission_ctx):
                    filtered.append(candidate)
            elif modality in ("document", "pdf"):
                if self._check_word_permission(candidate, user, permission_ctx):
                    filtered.append(candidate)
            else:
                # 图片/视频/链接: 字段级权限不适用，直接通过
                filtered.append(candidate)
        
        return filtered
    
    def _check_excel_permission(self,
                                 candidate: Candidate,
                                 user: User,
                                 ctx: PermissionContext) -> bool:
        """Excel 字段级权限校验"""
        
        # 1. 获取 Chunk 涉及的列
        involved_columns = candidate.metadata.get("involved_columns", [])
        if not involved_columns:
            # 无列信息时保守通过（或根据策略拒绝）
            return True
        
        # 2. 获取用户对该文档的字段权限
        doc_id = candidate.doc_id
        field_perm = ctx.field_permissions.get(doc_id, {})
        
        # 3. 获取工作表权限配置
        sheet_name = candidate.metadata.get("sheet_name", "")
        sheet_perms = field_perm.get("sheet_permissions", {})
        sheet_config = sheet_perms.get(sheet_name, {})
        
        # 4. 检查工作表级别
        access_level = sheet_config.get("access_level", "FULL")
        if access_level == "NONE":
            return False
        if access_level == "FULL":
            return True
        
        # 5. PARTIAL 模式: 检查列权限
        allowed_columns = set(sheet_config.get("allowed_columns", []))
        denied_columns = set(sheet_config.get("denied_columns", []))
        
        # 用户必须拥有所有涉及列的权限
        for col in involved_columns:
            # 检查黑名单
            if col in denied_columns:
                return False
            # 检查白名单（如果有白名单，必须在其中）
            if allowed_columns and col not in allowed_columns:
                return False
        
        # 6. 行范围检查（如有）
        row_range = candidate.metadata.get("row_range")
        allowed_row_range = sheet_config.get("allowed_row_range")
        if allowed_row_range and row_range:
            if row_range[0] < allowed_row_range[0] or row_range[1] > allowed_row_range[1]:
                return False
        
        return True
    
    def _check_word_permission(self,
                                candidate: Candidate,
                                user: User,
                                ctx: PermissionContext) -> bool:
        """Word 段落级权限校验"""
        
        doc_id = candidate.doc_id
        field_perm = ctx.field_permissions.get(doc_id, {})
        
        # 1. 章节权限检查
        section_headings = candidate.metadata.get("section_headings", [])
        denied_headings = set(field_perm.get("denied_headings", []))
        
        for heading in section_headings:
            if heading in denied_headings:
                return False
        
        # 2. 段落范围检查
        paragraph_range = candidate.metadata.get("paragraph_range")
        allowed_range = field_perm.get("allowed_paragraph_range")
        
        if allowed_range and paragraph_range:
            # Chunk 的段落范围必须完全在允许范围内
            if (paragraph_range[0] < allowed_range[0] or 
                paragraph_range[1] > allowed_range[1]):
                return False
        
        # 3. 允许的章节检查（白名单模式）
        allowed_sections = field_perm.get("allowed_sections", [])
        chunk_sections = candidate.metadata.get("section_ids", [])
        
        if allowed_sections and chunk_sections:
            # 至少有一个章节在白名单中
            if not any(s in allowed_sections for s in chunk_sections):
                return False
        
        return True
```

### 8.2 `build_generation_context` 实现

```python
from typing import List, Dict, Any, Optional
from dataclasses import dataclass, field

@dataclass
class CitationInfo:
    """引用信息"""
    ref_id: int
    chunk_id: str
    doc_id: str
    doc_title: str
    source_type: str
    page_number: Optional[int] = None
    sheet_name: Optional[str] = None
    time_range: Optional[Tuple[float, float]] = None
    url: Optional[str] = None
    image_urls: Dict[str, str] = field(default_factory=dict)

@dataclass
class MultimediaRef:
    """多媒体引用"""
    ref_id: int
    type: str  # "image" | "video"
    thumbnail_url: str
    original_url: str
    description: str
    ocr_text: Optional[str] = None
    time_start: Optional[float] = None  # 视频开始时间(秒)
    time_end: Optional[float] = None    # 视频结束时间(秒)

@dataclass
class GenerationContext:
    """生成上下文"""
    context_text: str
    citation_map: List[CitationInfo]
    multimedia_refs: List[MultimediaRef]
    total_tokens: int
    chunk_count: int
    context_meta: Dict[str, Any] = field(default_factory=dict)

class ContextBuilder:
    """生成上下文组装器"""
    
    def __init__(self,
                 tokenizer,  # 用于token计数
                 max_context_tokens: int = 4000,
                 chunk_separator: str = "\n\n"):
        self.tokenizer = tokenizer
        self.max_context_tokens = max_context_tokens
        self.chunk_separator = chunk_separator
    
    def build_generation_context(self,
                                  candidates: List[Candidate],
                                  query: str) -> GenerationContext:
        """
        构建生成上下文
        
        Args:
            candidates: 权限后过滤后的候选列表
            query: 原始查询（用于token计数参考）
        
        Returns:
            GenerationContext 对象
        """
        # 1. 去重与分组
        grouped = self._deduplicate_and_group(candidates)
        
        # 2. 注入引用标记
        cited_chunks, citation_map = self._inject_citations(grouped)
        
        # 3. 多模态内容组装
        multimedia_refs = self._assemble_multimedia(cited_chunks)
        
        # 4. 权限感知压缩
        compressed = self._compress_context(cited_chunks, query)
        
        # 5. 构建最终文本
        context_text = self._build_text(compressed)
        
        # 6. Token 计数
        total_tokens = self._count_tokens(context_text)
        
        return GenerationContext(
            context_text=context_text,
            citation_map=citation_map,
            multimedia_refs=multimedia_refs,
            total_tokens=total_tokens,
            chunk_count=len(compressed),
            context_meta={
                "query": query,
                "source_chunk_count": len(candidates),
                "deduplicated_count": len(cited_chunks),
                "compressed_count": len(compressed),
            }
        )
    
    def _deduplicate_and_group(self, 
                               candidates: List[Candidate]) -> List[Candidate]:
        """去重并按文档分组排序"""
        
        # 按 doc_id 分组
        doc_groups: Dict[str, List[Candidate]] = {}
        for c in candidates:
            if c.doc_id not in doc_groups:
                doc_groups[c.doc_id] = []
            doc_groups[c.doc_id].append(c)
        
        # 每组内按原文位置排序
        result = []
        for doc_id in doc_groups:
            chunks = doc_groups[doc_id]
            # 按 metadata 中的位置信息排序
            chunks.sort(key=lambda x: (
                x.metadata.get("page_number", 0),
                x.metadata.get("paragraph_index", 0),
                x.metadata.get("row_index", 0),
            ))
            result.extend(chunks)
        
        # 简单去重：内容相似度检查（生产环境可用更精确的算法）
        deduped = []
        seen_contents = set()
        for c in result:
            content_hash = hash(c.content[:200])  # 取前200字符哈希
            if content_hash not in seen_contents:
                seen_contents.add(content_hash)
                deduped.append(c)
        
        return deduped
    
    def _inject_citations(self, 
                          candidates: List[Candidate]) -> Tuple[List[Candidate], 
                                                                 List[CitationInfo]]:
        """为每个候选注入引用标记"""
        
        cited = []
        citation_map = []
        
        for i, candidate in enumerate(candidates, start=1):
            # 构建引用信息
            meta = candidate.metadata
            citation = CitationInfo(
                ref_id=i,
                chunk_id=candidate.chunk_id,
                doc_id=candidate.doc_id,
                doc_title=meta.get("doc_title", meta.get("filename", "未知文档")),
                source_type=meta.get("modality", "document"),
                page_number=meta.get("page_number"),
                sheet_name=meta.get("sheet_name"),
                time_range=meta.get("time_range"),
                url=meta.get("source_url"),
                image_urls=meta.get("thumbnail_urls", {}),
            )
            citation_map.append(citation)
            
            # 在 Chunk 内容前添加引用标记
            candidate.content = f"[{i}] {candidate.content}"
            candidate.metadata["ref_id"] = i
            cited.append(candidate)
        
        return cited, citation_map
    
    def _assemble_multimedia(self, 
                             candidates: List[Candidate]) -> List[MultimediaRef]:
        """组装多媒体引用"""
        
        multimedia_refs = []
        
        for c in candidates:
            modality = c.metadata.get("modality", "document")
            ref_id = c.metadata.get("ref_id", 0)
            
            if modality == "image":
                ref = MultimediaRef(
                    ref_id=ref_id,
                    type="image",
                    thumbnail_url=c.metadata.get("thumbnail_urls", {}).get("256", ""),
                    original_url=c.metadata.get("image_url", ""),
                    description=c.content[:200],
                    ocr_text=c.metadata.get("ocr_text", ""),
                )
                multimedia_refs.append(ref)
            
            elif modality == "video":
                ref = MultimediaRef(
                    ref_id=ref_id,
                    type="video",
                    thumbnail_url=c.metadata.get("keyframe_url", ""),
                    original_url=c.metadata.get("video_url", ""),
                    description=c.content[:200],
                    time_start=c.metadata.get("time_start"),
                    time_end=c.metadata.get("time_end"),
                )
                multimedia_refs.append(ref)
        
        return multimedia_refs
    
    def _compress_context(self,
                          candidates: List[Candidate],
                          query: str) -> List[Candidate]:
        """权限感知上下文压缩"""
        
        # 1. 构建完整上下文文本
        full_text = self.chunk_separator.join(c.content for c in candidates)
        
        # 2. 检查 token 数
        tokens = self._count_tokens(full_text)
        
        if tokens <= self.max_context_tokens:
            return candidates
        
        # 3. 需要压缩：优先保留高相关度 Chunk
        # 按分数排序（已是重排序后的顺序）
        compressed = []
        current_tokens = 0
        
        for c in candidates:
            chunk_tokens = self._count_tokens(c.content)
            
            if current_tokens + chunk_tokens <= self.max_context_tokens:
                compressed.append(c)
                current_tokens += chunk_tokens
            else:
                # 剩余空间不足，尝试摘要替代（可选）
                remaining = self.max_context_tokens - current_tokens
                if remaining > 100:
                    summary = self._summarize_chunk(c, max_tokens=remaining)
                    if summary:
                        c.content = summary
                        compressed.append(c)
                break
        
        return compressed
    
    def _build_text(self, candidates: List[Candidate]) -> str:
        """构建最终上下文文本"""
        return self.chunk_separator.join(c.content for c in candidates)
    
    def _count_tokens(self, text: str) -> int:
        """计算 token 数"""
        # 使用 tokenizer 编码
        try:
            return len(self.tokenizer.encode(text))
        except:
            # 降级: 按字符估算
            return len(text) // 2
    
    def _summarize_chunk(self, 
                         candidate: Candidate, 
                         max_tokens: int) -> Optional[str]:
        """对 Chunk 进行摘要（降级策略）"""
        # 简单截断（生产环境可用轻量摘要模型）
        content = candidate.content
        estimated_chars = max_tokens * 2  # 粗略估算
        if len(content) > estimated_chars:
            return content[:estimated_chars] + "..."
        return content
```

| **命中率** | 70% | 80% | 85% | 90% | 至少1个相关文档在Top-10 |

### 9.5 降级策略

| 场景 | 降级措施 | 预期影响 |
|------|----------|----------|
| Stage 2 模型超时 | 回退到 Stage 1 结果 | 排序精度下降 ~10% |
| Milvus 不可达 | 仅使用 Meilisearch Sparse | 语义召回率下降 ~25% |
| Meilisearch 不可达 | 仅使用 Milvus Dense | 关键词精确匹配下降 ~20% |
| 两模型均不可达 | 返回最近热门文档 + 提示 | 完全丧失查询相关性 |
| GPU 资源不足 | Stage1 切 CPU 推理 | 延迟增加 3~5x |
| 权限服务超时 | 使用缓存权限（可能过期） | 安全策略可能滞后 |
